<script>
    const canvas = document.getElementById('drawingCanvas');
    const ctx = canvas.getContext('2d');
    let isDrawing = false;
    let currentMode = 'free';
    let brushColor = '#D4AF37';
    let brushSize = 5;
    let brushOpacity = 1;
    let segments = 8;
    let history = [];
    let historyStep = -1;
    let lastX = 0;
    let lastY = 0;

    ctx.fillStyle = 'white';
    ctx.fillRect(0, 0, canvas.width, canvas.height);
    saveHistory();

    const prompts = [
        "Release what no longer serves you",
        "Draw your current energy",
        "Express gratitude through color",
        "Create from your heart center"
    ];

    document.getElementById('dailyPrompt').textContent = prompts[Math.floor(Math.random() * prompts.length)];

    function closeWelcome() {
        document.getElementById('welcomeOverlay').classList.add('hidden');
    }

    document.querySelectorAll('.mode-btn').forEach(function(btn) {
        btn.addEventListener('click', function() {
            document.querySelectorAll('.mode-btn').forEach(function(b) {
                b.classList.remove('active');
            });
            this.classList.add('active');
            currentMode = this.getAttribute('data-mode');

            document.getElementById('mandalaSegments').style.display = currentMode === 'mandala' ? 'block' : 'none';
            document.getElementById('traceTemplates').style.display = currentMode === 'trace' ? 'block' : 'none';
            document.getElementById('coloringPages').style.display = currentMode === 'color' ? 'block' : 'none';
            canvas.style.cursor = currentMode === 'hand' ? 'grab' : 'crosshair';
        });
    });

    document.getElementById('segmentSelect').addEventListener('change', function() {
        segments = parseInt(this.value);
    });

    document.querySelectorAll('[data-template]').forEach(function(card) {
        card.addEventListener('click', function() {
            document.querySelectorAll('[data-template]').forEach(function(c) {
                c.classList.remove('active');
            });
            this.classList.add('active');
            drawTemplate();
        });
    });

    document.querySelectorAll('[data-coloring]').forEach(function(card) {
        card.addEventListener('click', function() {
            document.querySelectorAll('[data-coloring]').forEach(function(c) {
                c.classList.remove('active');
            });
            this.classList.add('active');
            drawColoringPage();
        });
    });

    document.getElementById('brushColor').addEventListener('input', function() {
        brushColor = this.value;
    });

    document.getElementById('brushSize').addEventListener('input', function() {
        brushSize = parseInt(this.value);
        document.getElementById('sizeValue').textContent = brushSize;
    });

    document.getElementById('brushOpacity').addEventListener('input', function() {
        brushOpacity = parseInt(this.value) / 100;
        document.getElementById('opacityValue').textContent = this.value;
    });

    document.querySelectorAll('.chakra-color').forEach(function(color) {
        color.addEventListener('click', function() {
            document.querySelectorAll('.chakra-color').forEach(function(c) {
                c.classList.remove('active');
            });
            this.classList.add('active');
            brushColor = this.getAttribute('data-color');
            document.getElementById('brushColor').value = brushColor;
        });
    });

    document.getElementById('playBtn').addEventListener('click', function() {
        const musicSelect = document.getElementById('musicSelect');
        const selectedMusic = musicSelect.value;
        
        if (!selectedMusic) {
            alert('Please select a music track first');
            return;
        }

        const musicText = musicSelect.options[musicSelect.selectedIndex].text;
        document.getElementById('nowPlaying').textContent = 'Playing: ' + musicText;
        this.textContent = 'Pause';
    });

    canvas.addEventListener('mousedown', startDrawing);
    canvas.addEventListener('mousemove', draw);
    canvas.addEventListener('mouseup', stopDrawing);
    canvas.addEventListener('mouseout', stopDrawing);

    canvas.addEventListener('touchstart', function(e) {
        e.preventDefault();
        const touch = e.touches[0];
        const rect = canvas.getBoundingClientRect();
        lastX = touch.clientX - rect.left;
        lastY = touch.clientY - rect.top;
        isDrawing = true;
    });

    canvas.addEventListener('touchmove', function(e) {
        e.preventDefault();
        if (!isDrawing) return;
        const touch = e.touches[0];
        const rect = canvas.getBoundingClientRect();
        const x = touch.clientX - rect.left;
        const y = touch.clientY - rect.top;
        drawLine(lastX, lastY, x, y);
        lastX = x;
        lastY = y;
    });

    canvas.addEventListener('touchend', function(e) {
        e.preventDefault();
        stopDrawing();
    });

    function startDrawing(e) {
        if (currentMode === 'hand') return;
        isDrawing = true;
        const rect = canvas.getBoundingClientRect();
        lastX = e.clientX - rect.left;
        lastY = e.clientY - rect.top;
    }

    function draw(e) {
        if (!isDrawing || currentMode === 'hand') return;
        const rect = canvas.getBoundingClientRect();
